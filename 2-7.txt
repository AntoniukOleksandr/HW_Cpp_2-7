//Використовуючи один із варіантів циклу і оператор break, написати програму, 
//в якій користувач вводитиме числові значення відмінні від нуля доти, поки не буде введено значення 0. 
//В результаті, має бути вихід з програми через оператор break.

#include <iostream>
using namespace std;
int main() {
    int NUM;

    while (true) {
        cout << "введіть число (0 - якщо хочете вийти з програми): ";
        cin >> NUM;

        if (NUM == 0) {
            cout << "ви ввели 0, закінчення і вихід із програми" << endl;
            break;
        }

        cout << "ввести число: " << NUM << endl;
    }

    return 0;
}
